# переменные
- именованная область памяти хранящая значение/именованная ячейка с информацией
```python
x = 10                     #int
k = 1.2                    #float
a = 'str'                  #string
s = true                   #bool
e = [1, 2, false, 'hello'] #list
d = {1, 2, false, 'hello'} #set
f = (1, 2, false, 'hello') #tuple
h = {'h' : 1, 'p' : 3.14}  #dictionary

text = input()
print(len(text))           #lenght 
print(text[0])             #1st symbol
for i in range(len(text)):
    if i % 2 == 0:
        print(text[i])     #even symbols
```
